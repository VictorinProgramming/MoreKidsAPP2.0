--> Quando Criado um novo Repositório <--

git init
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/VictorinProgramming/MoreKidsAPP2.0.git
git push -u origin main

--> Quando temos um Repositório Existente <--

git remote add origin https://github.com/VictorinProgramming/MoreKidsAPP2.0.git
git branch -M main
git push -u origin main

--> Criando novas Branchs <--

git branch staging dev

--> Usando a Branch <--
git checkout staging dev
